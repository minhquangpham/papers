\frametitle{Revisited methods}
\begin{itemize}
	\item Domain control (Kobus et al 2017)
	\begin{itemize}
		\item Domain tag (\colorbox{green}{DC-Tag})
		\item Domain embedding (\colorbox{green}{DC-Feat})
	\end{itemize}
	\item Lexicalized Domain Representation (Pham et al 2019) (\colorbox{green}{LDR})
	\item Domain mixing (Britz et al 2018)
	\begin{itemize}
		\item Domain tag in target (\colorbox{green}{TTM})
		\item Domain mixing (\colorbox{green}{DM})
		\item Adversarial Domain mixing (\colorbox{green}{ADM})
	\end{itemize}
	\item Residual Adapter (Bapna et al 2019)
	\begin{itemize}
		\item Fine-tuning (\colorbox{green}{FT-Res})
		\item Multi-task learning (\colorbox{green}{MDL-Res})
	\end{itemize}
	\item Generic (\colorbox{green}{Mixed})
	\item Full fine-tuning (\colorbox{green}{FT-full})
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Recommendations for future MDMT studies}
\begin{itemize}	
	\item Be explicit about what Multi-domain means, and whether domain tag are known in train and test.
	\item Clarify the objective, wrt test distributions, design experiments accordingly.	
	\item Use as many domains as possible.
	\item Pay attention to the balance between domains, wrt their similarities.
	
	
\end{itemize}
